    "    df = build_df(ytime, ystatus, mtime0, mtime, mstatus)\n",

    "    med = Mediation(\n",

    "        \"exp\",\n",

    "        outcome_predict_kwargs={\"pred_only\": True},\n",
    "    )\n",
    "    med_result = med.fit(n_rep=20)\n",
